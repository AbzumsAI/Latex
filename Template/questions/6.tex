فرض کنید $A, B, C \subseteq M$. هر یک از موارد زیر را ثابت یا با مثال نقض رد کنید.

\begin{enumerate}
	\item[] $A - C = B - C \implies A = B$
	\item[] $[(A \cap C = B \cap C) \land (A - C = B - C)] \implies A = B$
	\item[] $[(A \cup C = B \cup C) \land (A - C = B - C)] \implies A = B$
	\item[] $P(A - B) = P(A) - P(B)$
\end{enumerate}
